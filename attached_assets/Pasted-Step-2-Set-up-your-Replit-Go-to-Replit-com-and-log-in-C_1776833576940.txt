import streamlit as st
import pandas as pd

# The Links go here
RETENTION_URL = "PASTE_YOUR_RETENTION_URL_HERE"
NSF_URL = "PASTE_YOUR_NSF_URL_HERE"

st.set_page_config(layout="wide")
st.title("🚀 Agent Performance Dashboard")

@st.cache_data(ttl=7200)
def load_data():
    ret_df = pd.read_csv(RETENTION_URL)
    nsf_df = pd.read_csv(NSF_URL)
    return ret_df, nsf_df

retention_df, nsf_df = load_data()

tab1, tab2 = st.tabs(["Retention Team", "NSF Team"])

with tab1:
    st.header("Retention Team")
    st.dataframe(retention_df, use_container_width=True)

with tab2:
    st.header("NSF Team")
    st.dataframe(nsf_df, use_container_width=True)
Important: Find where it says PASTE_YOUR_RETENTION_URL_HERE and PASTE_YOUR_NSF_URL_HERE in the code and replace those with the special links you created in Step 1 (make sure to keep the quote marks around the links).
